// script.js document.addEventListener("DOMContentLoaded", function () { const loginForm = document.querySelector("form"); loginForm.addEventListener("submit", function (event) { event.preventDefault(); // Prevent the form from submitting the traditional way const username = loginForm.username.value.trim(); const password = loginForm.password.value.trim(); if (username === "" || password === "") { alert("Please enter both username and password."); } else { alert(`Login successful! \nUsername: ${username} \nPassword: ${password}`); // Here you can add your logic to handle the form submission, like sending the data to a server } }); });
